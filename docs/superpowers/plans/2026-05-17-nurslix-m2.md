### Task M2.2: "今天答錯了什麼" 自動素材抽取

- [ ] **Step 1: Build wrong-answer extractor**

`lib/agents/marketing/myMistakeExtractor.ts`:

```typescript
import { prisma } from "@/lib/prisma";

const ADMIN_EMAIL = "cmshj30326@gmail.com"; // owner = self

export async function getTodaysMistakes(maxCount = 3) {
  const user = await prisma.user.findUnique({ where: { email: ADMIN_EMAIL }, select: { id: true } });
  if (!user) return [];
  const todayStart = new Date();
  todayStart.setHours(0, 0, 0, 0);
  const answers = await prisma.userAnswer.findMany({
    where: { userId: user.id, answeredAt: { gte: todayStart }, isCorrect: false },
    orderBy: { answeredAt: "desc" },
    take: maxCount,
    include: {
      question: {
        select: { id: true, stem: true, stemZh: true, correctAnswer: true, rationale: true, rationaleZh: true, domain: true },
      },
    },
  });
  return answers.map(a => ({
    questionId: a.questionId,
    stem: a.question.stem,
    correctAnswer: a.question.correctAnswer,
    yourAnswer: a.selectedAnswer,
    rationale: a.question.rationaleZh ?? a.question.rationale,
    domain: a.question.domain,
  }));
}
```

import { NextRequest, NextResponse } from "next/server";
import { prisma } from "@/lib/prisma";
import { requireAdmin } from "@/lib/admin";

export async function POST(req: NextRequest, { params }: { params: { id: string } }) {
  const guard = await requireAdmin();
  if (guard instanceof NextResponse) return guard;
  const { url } = await req.json().catch(() => ({}));
  await prisma.marketingContent.update({
    where: { id: params.id },
    data: { status: "published", publishedAt: new Date(), publishedUrl: url ?? null },
  });
  return NextResponse.json({ ok: true });
}

### Task M2.8: "English reading ability" dashboard widget

**Background:** Per spec 4.5.3 — show users their progress in reading English NCLEX questions. Concrete metric: % of questions where user reads the stem in < 30s (proxy for fluent reading).

**Files:**
- Create: `app/api/user/english-ability/route.ts`
- Modify: dashboard page to add the widget

- [ ] **Step 1: API computes reading ability**

```typescript
import { NextResponse } from "next/server";
import { auth } from "@/lib/auth";
import { prisma } from "@/lib/prisma";

export async function GET() {
  const session = await auth();
  if (!session?.user?.id) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  const ans = await prisma.userAnswer.findMany({
    where: { userId: session.user.id, timeSpentSec: { not: null } },
    select: { timeSpentSec: true, isCorrect: true },
    take: 200,
    orderBy: { answeredAt: "desc" },
  });
  const fluentReads = ans.filter(a => (a.timeSpentSec ?? 99999) < 30).length;
  const total = ans.length;
  const fluentPct = total > 0 ? Math.round((fluentReads / total) * 100) : 0;
  return NextResponse.json({
    totalAnswered: total,
    fluentReadPercent: fluentPct,
    label: fluentPct >= 80 ? "流暢" : fluentPct >= 50 ? "中等" : "需練習",
  });
}
```
