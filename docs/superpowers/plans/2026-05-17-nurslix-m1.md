```typescript
/**
 * GET /api/cron/close-stale-reports
 *
 * One-shot cleanup: bulk-marks pre-existing IN_REVIEW QuestionReports as
 * RESOLVED_INVALID where the AI triage already concluded LIKELY_INVALID
 * but admin never closed them out. Guarded by CRON_SECRET.
 *
 * Safe to re-run — idempotent on already-resolved rows.
 */
import { NextRequest, NextResponse } from "next/server";
import { prisma } from "@/lib/prisma";

export const dynamic = "force-dynamic";

export async function GET(req: NextRequest) {
  const secret = process.env.CRON_SECRET;
  if (!secret) return NextResponse.json({ error: "CRON_SECRET not configured" }, { status: 500 });
  const auth = req.headers.get("authorization") ?? "";
  if (auth !== `Bearer ${secret}`) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

  const result = await prisma.questionReport.updateMany({
    where: {
      status: "IN_REVIEW",
      triageVerdict: "LIKELY_INVALID",
    },
    data: {
      status: "RESOLVED_INVALID",
      resolvedAt: new Date(),
      resolvedBy: "cron:close-stale-reports",
      resolution: "Triaged as LIKELY_INVALID by AI; bulk-closed during M1 cleanup.",
    },
  });

  return NextResponse.json({ ok: true, closed: result.count });
}
```

```typescript
"use client";

import { useState, useEffect } from "react";
import { Languages } from "lucide-react";

const STORAGE_KEY = "nurslix-bilingual-mode";

export function useBilingualMode() {
  const [mode, setMode] = useState<"en" | "zh" | "both">("both");

  useEffect(() => {
    if (typeof window === "undefined") return;
    const stored = window.localStorage.getItem(STORAGE_KEY);
    if (stored === "en" || stored === "zh" || stored === "both") {
      setMode(stored);
    }
  }, []);

  const updateMode = (m: "en" | "zh" | "both") => {
    setMode(m);
    if (typeof window !== "undefined") {
      window.localStorage.setItem(STORAGE_KEY, m);
    }
  };

  return { mode, setMode: updateMode };
}

interface ToggleProps {
  mode: "en" | "zh" | "both";
  onChange: (m: "en" | "zh" | "both") => void;
}

export default function BilingualToggle({ mode, onChange }: ToggleProps) {
  return (
    <div className="inline-flex items-center gap-1 bg-[var(--bg-elevated)] border border-[var(--border-subtle)] rounded-lg p-1">
      <Languages size={14} className="text-[var(--text-muted)] ml-1" />
      {(["en", "both", "zh"] as const).map((m) => (
        <button
          key={m}
          type="button"
          onClick={() => onChange(m)}
          className={`px-2 py-1 text-xs rounded transition-colors ${
            mode === m
              ? "bg-[var(--gold)] text-[#080E1A] font-semibold"
              : "text-[var(--text-secondary)] hover:text-[var(--text-primary)]"
          }`}
          aria-pressed={mode === m}
        >
          {m === "en" ? "EN" : m === "zh" ? "中" : "中/EN"}
        </button>
      ))}
    </div>
  );
}
```

```typescript
export const dynamic = "force-dynamic";

export async function POST(req: NextRequest) {
  const { words } = await req.json() as { words: string[] };
  if (!Array.isArray(words) || words.length === 0 || words.length > 50) {
    return NextResponse.json({ error: "Invalid words list" }, { status: 400 });
```

- [ ] **Step 3: Build `WordHover.tsx`**

A component that takes a string of English text and wraps each word in a hoverable span. On hover, calls the lookup API (debounced + cached) and shows a tooltip.

```typescript
"use client";

import { useState, useRef, useEffect } from "react";

interface VocabInfo {
  word: string;
  partOfSpeech: string | null;
  definitionZh: string | null;
  nclexUsage: string | null;
}

const cache = new Map<string, VocabInfo | null>();

async function lookup(words: string[]): Promise<Record<string, VocabInfo>> {
  const uncached = words.filter(w => !cache.has(w.toLowerCase()));
  if (uncached.length > 0) {
    const res = await fetch("/api/vocab/lookup", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ words: uncached }),
    });
    if (res.ok) {
      const { results }: { results: VocabInfo[] } = await res.json();
      const seen = new Set(results.map(r => r.word.toLowerCase()));
      for (const r of results) cache.set(r.word.toLowerCase(), r);
      for (const w of uncached) if (!seen.has(w.toLowerCase())) cache.set(w.toLowerCase(), null);
    }
  }
  const out: Record<string, VocabInfo> = {};
  for (const w of words) {
    const info = cache.get(w.toLowerCase());
    if (info) out[w] = info;
  }
  return out;
}

export default function WordHover({ text }: { text: string }) {
  const tokens = text.split(/(\s+)/);
  const [hoveredWord, setHoveredWord] = useState<string | null>(null);
  const [info, setInfo] = useState<VocabInfo | null>(null);
  const debounceRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  useEffect(() => {
    if (!hoveredWord) {
      setInfo(null);
      return;
    }
    if (debounceRef.current) clearTimeout(debounceRef.current);
    debounceRef.current = setTimeout(async () => {
      const result = await lookup([hoveredWord]);
      setInfo(result[hoveredWord] ?? null);
    }, 200);
    return () => {
      if (debounceRef.current) clearTimeout(debounceRef.current);
    };
  }, [hoveredWord]);

  return (
    <span className="relative">
      {tokens.map((tok, i) => {
        const isWord = /^[A-Za-z]+$/.test(tok);
        if (!isWord) return <span key={i}>{tok}</span>;
        return (
          <span
            key={i}
            className="hover:bg-[var(--gold-dim)] cursor-help rounded px-0.5"
            onMouseEnter={() => setHoveredWord(tok)}
            onMouseLeave={() => setHoveredWord(null)}
          >
            {tok}
            {hoveredWord === tok && info && (
              <span className="absolute top-full left-0 mt-1 bg-[var(--bg-surface)] border border-[var(--border-default)] rounded-lg p-2 text-xs z-50 min-w-[200px] shadow-lg">
                <span className="font-semibold">{info.word}</span>
                {info.partOfSpeech && <span className="ml-2 text-[var(--text-muted)]">{info.partOfSpeech}</span>}
                {info.definitionZh && <div className="mt-1">{info.definitionZh}</div>}
                {info.nclexUsage && <div className="mt-1 text-[var(--text-muted)]">NCLEX: {info.nclexUsage}</div>}
              </span>
            )}
          </span>
        );
      })}
    </span>
  );
}
```
